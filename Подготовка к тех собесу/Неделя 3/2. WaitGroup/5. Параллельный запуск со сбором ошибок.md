Реализуйте функцию `` `execute` ``, которая параллельно запускает переданные задачи и возвращает массив ошибок из тех задач, которые завершились с ошибкой.

Требования:

- Каждая задача запускается в отдельной горутине
- Функция возвращает все ошибки от неуспешных задач
- Порядок ошибок в результате не важен

Данную задачу предлагается попробовать решить двумя способами: используя `WaitGroup` и без неё (через `каналы`).

```go
package main

type Task interface {

    Run() error

}

func execute(tasks []Task) []error {

    // TODO

}
```

### Тесты

```go

package main

import (
    "fmt"
    "os"
    "time"
	"testing"
)

type TestCase struct {
    name  string
    prepare func() *prepareData
    check func(*prepareData) bool
}

type prepareData struct {
    tasks []Task
    wantErr  bool
    wantLen  int
}

var testCases = []TestCase{
    {
        name: "все задачи успешны",
        prepare: func() *prepareData {
            return &prepareData{
                tasks: []Task{&MockTask{result: nil}, &MockTask{result: nil}},
                wantErr: false,
                wantLen: 0,
            }
        },
        check: func(data *prepareData) bool {
            errors := execute(data.tasks)

            if len(errors) != data.wantLen {
                _, _ = fmt.Fprintf(
                    os.Stderr, "execute() вернул %d ошибок, ожидалось %d\n",
                    len(errors), data.wantLen)

                    return false
            }

            if data.wantErr && len(errors) == 0 {

                _, _ = fmt.Fprintf(
                    os.Stderr, "ожидались ошибки, но не получили ни одной\n")
                    return false

            }

            if !data.wantErr && len(errors) > 0 {
                _, _ = fmt.Fprintf(
                    os.Stderr, "не ожидались ошибки, но пришло %d\n", len(errors))

                    return false

            }
            return true
        },
    },
}

// MockTask для тестирования
type MockTask struct {
    result error
    delay  time.Duration
}

func (m *MockTask) Run() error {
    time.Sleep(m.delay)
    return m.result
}

func TestSolution(t *testing.T) {
	for _, tc := range testCases {
		t.Run(tc.name, func(t *testing.T) {
			obj := tc.prepare()
			if !tc.check(obj) {
				t.Errorf("тест не пройден")
			}
		})
	}
}
```

### Решение с WaitGroup

```go

package main

import "sync"

type Task interface {
    Run() error
}

func execute(tasks []Task) []error {
    group := &sync.WaitGroup{}
    group.Add(len(tasks))
    errChan := make(chan error, len(tasks))
    for _, task := range tasks {
        go func(task Task) {
            defer group.Done()
            if err := task.Run(); err != nil {
                errChan <- err
            }
        }(task)
    }

    group.Wait()
    close(errChan)

    var res []error
    for err := range errChan {
        res = append(res, err)
    }

    return res
}
```
### Решение с chan

```go

package main

type Task interface {
    Run() error
}

func execute(tasks []Task) []error {
    errCh := make(chan error, len(tasks))

    for _, t := range tasks {
        go func(t Task) {
            errCh <- t.Run()
        }(t)
    }

    res := make([]error, 0, len(tasks))

    for i := 0; i < len(tasks); i++ {
        if err := <-errCh; err != nil {
            res = append(res, err)
        }
    }

    return res
}
```
