Представьте, что ваш коллега принес вам на ревью новую задачу и просит поревьюить ее. В задаче он добавляет новую функцию для запуска заданий, которые уже готовы к запуску. Те, что не готовы, нужно просто пропустить.

Функция `task.Run()` может паниковать и в этом случае нужно сделать `recovery` для задачи, которая паникует, а остальные нужно выполнить и завершить программу в штатном режиме.

```go
type Task interface {
    IsReady() bool
    Run()
}

func RunTasks(tasks []Task) {
    wg := sync.WaitGroup{}

    for _, task := range tasks {
        if !task.IsReady() {
            continue
        }

        go func() {
            wg.Add(1)
            task.Run()
            wg.Done()
        }()
    }
}

```
В коде есть несколько ошибок, найдите их и исправьте.

### Решение

```go
package main

import "sync"

type Task interface {
    IsReady() bool
    Run()
}

func RunTasks(tasks []Task) {
    wg := sync.WaitGroup{}

    for _, task := range tasks {
        if !task.IsReady() {
            continue
        }

        wg.Add(1)
        go func() {
            defer func() {
                _ = recover()
            }()
            defer wg.Done()
            task.Run()
        }()
    }

    wg.Wait()
}
```
