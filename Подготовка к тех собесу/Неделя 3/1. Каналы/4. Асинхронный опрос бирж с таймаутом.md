Перед вами код, который опрашивает несколько бирж. Сейчас он работает синхронно.

Нужно реализовать функцию, которая параллельно опрашивает несколько бирж и возвращает только те результаты, которые успели прийти **за 2 секунды**. Время в 2 секунды ограничивает **всю операцию целиком**, а не каждый отдельный запрос. Используйте `context` для прерывания ожидания.

```go
type Backend interface {
    DoRequest() string
}

func DoRequests(backends []Backend) []string {
    results := make([]string, 0, len(backends))
    for _, backend := range backends {
        results = append(results, backend.DoRequest())
    }

    return results
}

```

```go
type mockBackend struct {  
    response string

func ContainsAll(results []string, values ...string) bool {  
    m := make(map[string]bool)  
    for _, r := range results {  
       m[r] = true  
    }  
    for _, v := range values {  
       if !m[v] {  
          return false  
       }  
    }  
    return true
```

### Решение

```go
package main

import (
    "context"
    "time"
	"testing"
)

type Backend interface {
    DoRequest() string
}

func DoRequests(backends []Backend) []string {
    ctx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
    defer cancel()

    ch := make(chan string, len(backends))

    for _, backend := range backends {
        go func() {
            ch <- backend.DoRequest()
        }()
    }

    results := make([]string, 0, len(backends))

    for range len(backends) {
        select {
        case result := <-ch:
            results = append(results, result)
        case <-ctx.Done():
            return results
        }
    }

    return results
}
```
