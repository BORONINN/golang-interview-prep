Необходимо написать функцию `` `WriteHelloWorld` ``, которая записывает строку `` `"Hello world"` `` в переданный `` `io.Writer` ``. Если запись не удалась, то функция возвращает ошибку.

```go
package main

import (

    "io"

)

func WriteHelloWorld(out io.Writer) error {

    return nil

}
```
---
### Тесты
```go
package main

import (
    "bytes"
    "errors"
    "io"
	"testing"
)

type TestCase struct {
    name  string
    input io.Writer
    check func(writer io.Writer, err error) bool
}

var testCases = []TestCase{
    {
        name:  "Успешная запись в буфер",
        input: &bytes.Buffer{},
        check: func(writer io.Writer, err error) bool {
            if err != nil {
                return false
            }
            buf := writer.(*bytes.Buffer)
            return buf.String() == "Hello world"
        },
    },
    {
        name:  "Ошибка записи в writer",
        input: &errorWriter{},
        check: func(writer io.Writer, err error) bool {
            return errors.Is(err, testError)
        },
    },
    {
        name:  "Ошибка при nil интерфейсе",
        input: nil,
        check: func(writer io.Writer, err error) bool {
            return err != nil
        },
    },
}

var testError = errors.New("write error")

// Вспомогательная структура для эмуляции ошибки записи
type errorWriter struct{}

func (w *errorWriter) Write(p []byte) (n int, err error) {
    return 0, testError
}

func TestSolution(t *testing.T) {
	for _, tc := range testCases {
		t.Run(tc.name, func(t *testing.T) {
			err := WriteHelloWorld(tc.input)
			if !tc.check(tc.input, err) {
				t.Errorf("тест не пройден")
			}
		})
	}
}
```

### Решение

```go

package main

import (
    "errors"
    "io"
    "reflect"
)

func WriteHelloWorld(out io.Writer) error {
    if out == nil || reflect.ValueOf(out).IsZero() {
        return errors.New("nil interface")
    }

    _, err := out.Write([]byte("Hello world"))
    return err
}
```
