легко

Даны две ссылки `val1` и `val2` на целые числа. Нужно реализовать функцию `` `Swap` ``, которая меняет их значения местами.

Считать, что указатель не может быть `nil`.

**Пример:**

Ввод: val1 = 20, val2 = 10; Swap(&val1, &val2)
Вывод: val1 = 10, val2 = 20

---
### Тесты

```go

package main


import "testing"
type TestCase struct {
    name     string
    input    tuple
    expected tuple
}

var testCases = []TestCase{
    {
        name: "Два разных числа",
        input: tuple{
            value1: 1,
            value2: 2,
        },
        expected: tuple{
            value1: 2,
            value2: 1,
        },
    },
    {
        name: "Два одинаковых числа",
        input: tuple{
            value1: 2,
            value2: 2,
        },
        expected: tuple{
            value1: 2,
            value2: 2,
        },
    },
    {
        name: "Числа разных знаков",
        input: tuple{
            value1: -1,
            value2: 2,
        },
        expected: tuple{
            value1: 2,
            value2: -1,
        },
    },
    {
        name: "Отрицательные числа",
        input: tuple{
            value1: -1,
            value2: -2,
        },
        expected: tuple{
            value1: -2,
            value2: -1,
        },
    },
}

type tuple struct {
    value1 int
    value2 int
}

func TestSolution(t *testing.T) {
	for _, tc := range testCases {
		t.Run(tc.name, func(t *testing.T) {
			a, b := tc.input.value1, tc.input.value2
			Swap(&a, &b)
			if a != tc.expected.value1 || b != tc.expected.value2 {
				t.Errorf("got (%v, %v), want (%v, %v)", a, b, tc.expected.value1, tc.expected.value2)
			}
		})
	}
}
```

---
### Решение

```go

package main

func Swap(val1 *int, val2 *int) {
    *val1, *val2 = *val2, *val1
}
```
