легко

Дан массив целых чисел `` `values` и целое число `chunkSize` `` размер чанка. Необходимо написать функцию, которая разбивает исходный массив на подмассивы (чанки) указанного размера.

Последний чанк может содержать меньше `chunkSize` элементов - остаток.

**Пример 1:**

Ввод: values = [1, 2, 3, 4, 5], chunkSize = 2
Вывод: [[1, 2], [3, 4], [5]]

**Пример 2:**

Ввод: values = [1, 2, 3, 4, 5], chunkSize = 0
Вывод: []

**Ограничения:**

- `chunkSize >= 0`
- `len(values) >= 0`
---

### Тесты

```go
package main


import "testing"
type TestCase struct {
    input    []int
    chunkLen int
    expected [][]int
    name     string
}

var testCases = []TestCase{
    {
        name:     "Размер чанка 2",
        input:    []int{1, 2, 3, 4},
        chunkLen: 2,
        expected: [][]int{
            {1, 2},
            {3, 4},
        },
    },
    {
        name:     "Размер чанка 3",
        input:    []int{1, 2, 3, 4, 5, 6, 7, 8, 9},
        chunkLen: 3,
        expected: [][]int{
            {1, 2, 3},
            {4, 5, 6},
            {7, 8, 9},
        },
    },
    {
        name:     "Размер чанка 1",
        input:    []int{1, 2, 3, 4},
        chunkLen: 1,
        expected: [][]int{
            {1}, {2}, {3}, {4},
        },
    },
    {
        name:     "Размер чанка равен размеру слайса",
        input:    []int{1, 2, 3, 4},
        chunkLen: 4,
        expected: [][]int{
            {1, 2, 3, 4},
        },
    },
}

func TestSolution(t *testing.T) {
	for _, tc := range testCases {
		t.Run(tc.name, func(t *testing.T) {
			result := Chunk(tc.input, tc.chunkLen)
			if len(result) != len(tc.expected) {
				t.Errorf("got %v, want %v", result, tc.expected)
				return
			}
			for i := range result {
				if len(result[i]) != len(tc.expected[i]) {
					t.Errorf("got %v, want %v", result, tc.expected)
					return
				}
				for j := range result[i] {
					if result[i][j] != tc.expected[i][j] {
						t.Errorf("got %v, want %v", result, tc.expected)
						return
					}
				}
			}
		})
	}
}
```
---
### Решение

```go
package main

func Chunk(values []int, chunkSize int) [][]int {
    if chunkSize < 1 || len(values) == 0 {
        return [][]int{}
    }

    var chunks [][]int

    for chunkSize < len(values) {
        values, chunks = values[chunkSize:], append(chunks, values[0:chunkSize:chunkSize])
    }
    return append(chunks, values)
}
```
