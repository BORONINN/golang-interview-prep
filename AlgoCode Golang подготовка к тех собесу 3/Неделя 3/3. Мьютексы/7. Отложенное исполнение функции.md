## Решение
```go
package main

import (
    "sync"
    "time"
)

type Scheduler struct {
    timers map[int]*time.Timer
    mu     sync.Mutex
}

// создает новый планировщик
func NewScheduler() *Scheduler {
    return &Scheduler{
        timers: make(map[int]*time.Timer),
    }
}

func (s *Scheduler) Delay(key int, f func(), t time.Duration) {
    s.mu.Lock()
    defer s.mu.Unlock()

    // если таймер уже существует для этого ключа — останавливаем
    if existingTimer, exists := s.timers[key]; exists {
        existingTimer.Stop()
        delete(s.timers, key)
    }

    // если f == nil, ничего не делаем
    if f == nil {
        return
    }

    // сохраняем ссылку на таймер в замыкании для защиты от гонки
    var timer *time.Timer
    timer = time.AfterFunc(t, func() {
        // выполняем функцию
        f()
        // удаляем только если в map всё ещё наш таймер
        s.mu.Lock()
        if s.timers[key] == timer {
            delete(s.timers, key)
        }
        s.mu.Unlock()
    })
    s.timers[key] = timer
}
```
