Функция должна возвращать канал, в который будут последовательно отправляться все элементы из исходного слайса, а так же **закрыть канал**, когда все значения будут записаны.

func SliceToChannel(values []int) <-chan int {

    return nil

}

### Тесты
```go
package main

import "testing"
type TestCase struct {
    name  string
    input []int
}

var testCases = []TestCase{
    {
        name:  "Пустой слайс",
        input: []int{},
    },
    {
        name:  "Слайс с одним элементом",
        input: []int{42},
    },
    {
        name:  "Слайс с несколькими элементами",
        input: []int{1, 2, 3, 4, 5},
    },
    {
        name:  "Слайс с отрицательными числами",
        input: []int{-1, -2, -3},
    },
    {
        name:  "Слайс с нулевыми значениями",
        input: []int{0, 0, 0},
    },
}

func TestSolution(t *testing.T) {
	for _, tc := range testCases {
		t.Run(tc.name, func(t *testing.T) {
			ch := SliceToChannel(tc.input)
			var result []int
			for v := range ch {
				result = append(result, v)
			}
			if len(result) != len(tc.input) {
				t.Errorf("got %v, want %v", result, tc.input)
				return
			}
			for i := range result {
				if result[i] != tc.input[i] {
					t.Errorf("got %v, want %v", result, tc.input)
					return
				}
			}
		})
	}
}
```


---
### Решение

```go

package main

func SliceToChannel(values []int) <-chan int {
    ch := make(chan int)

    go func() {
        for _, value := range values {
            ch <- value
        }
        close(ch)
    }()

    return ch
}
```
