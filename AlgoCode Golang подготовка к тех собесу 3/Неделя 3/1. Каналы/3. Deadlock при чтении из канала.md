В функции `PrintFive` происходит `deadlock`.

```go
func PrintFive() {
    ch := make(chan int)

    go func() {
        for i := 0; i < 5; i++ {
            ch <- i
        }
    }()

    for n := range ch {
        fmt.Println(n)
    }
}

```
Исправьте код, чтобы избежать блокировки.

```go
package main

func PrintFive() {

    return nil

}
```

### Тесты
```go
package main


import "testing"
type TestCase struct {
    name     string
    expected string
}

var testCases = []TestCase{
    {
        name:     "Проверка вывода",
        expected: "0\n1\n2\n3\n4\n",
    },
}

func TestSolution(t *testing.T) {  
    for _, tc := range testCases {  
       t.Run(tc.name, func(t *testing.T) {  
          old := os.Stdout  
          r, w, _ := os.Pipe()  
          os.Stdout = w  
  
          PrintFive()  
  
          w.Close()  
          var buf bytes.Buffer  
          buf.ReadFrom(r)  
          os.Stdout = old  
  
          result := buf.String()  
          if result != tc.expected {  
             t.Errorf("got %q, want %q", result, tc.expected)  
          }  
       })  
    }  
}
```

### Решение

```go

package main

import "fmt"

func PrintFive() {
    ch := make(chan int)

    go func() {
        for i := 0; i < 5; i++ {
            ch <- i
        }
        close(ch)
    }()

    for n := range ch {
        fmt.Println(n)
    }
}
```
