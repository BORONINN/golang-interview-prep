Дан массив функций `` `func()` ```.` Напишите функцию `` `ConcurrentlyRun` ``, которая:

1. Принимает массив функций
2. Запускает каждую в отдельной горутине
3. В случае успешной отработки каждой функции `ConcurrentlyRun` возвращает `` `nil`. `` В случае, если что-то идет не так (например паника), возвращает ошибку.
```go
package main

func ConcurrentlyRun(callback ...func()) error {

    return nil

}
```

                func() { counter.Add(1) },

        },  
    },

        name: "Проверка на пустой список функций",  
        check: func() bool {  
            err := ConcurrentlyRun()  
            return err == nil

### Решение
```go

package main  
  
import (  
    "fmt"  
    "sync"  
    "sync/atomic"  
)  
  
func ConcurrentlyRun(callback ...func()) error {  
    wg := sync.WaitGroup{}  
    wg.Add(len(callback))  
  
    panicErr := atomic.Pointer[error]{}  
  
    for _, fn := range callback {  
        fn := fn  
        go func() {  
            defer wg.Done()  
            defer func() {  
                if r := recover(); r != nil {  
                    if err, ok := r.(error); ok {  
                        panicErr.Store(&err)  
                    } else {  
                        err := fmt.Errorf("%v", err)  
                        panicErr.Store(&err)  
                    }  
                }  
            }()  
  
            fn()  
        }()  
    }  
  
    wg.Wait()  
  
    if err := panicErr.Load(); err != nil {  
        return *err  
    }  
    return nil  
}
```
