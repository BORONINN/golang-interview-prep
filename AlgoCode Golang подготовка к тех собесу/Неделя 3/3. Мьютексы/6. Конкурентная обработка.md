У нас есть система обработки событий для заявок. Каждое событие содержит уникальный идентификатор заявки (`` `id` ``). Несколько рабочих `` `консьюмеров` `` могут одновременно обрабатывать разные события.

- События для одной и той же заявки могут поступать несколько раз подряд
- Каждое событие обрабатывается цепочкой обработчиков (`` `handlers` ``)
- Разные заявки должны обрабатываться параллельно для эффективности

Нужно предотвратить ситуацию, когда несколько рабочих одновременно обрабатывают события одной и той же заявки, так как это может привести к конфликтам.

То есть необходимо реализовать механизм блокировки, который гарантирует, что для каждой конкретной заявки в любой момент времени работает только один консьюмер.

Программа использующая `Subscriber` могла бы выглядеть так:

```go
func main() {
	// Создаем подписчик с цепочкой обработчиков
	subscriber := NewSubscriber([]Handler{
		handler1,
		handler2,
		handler3,
	})

	// ID заявок для обработки (некоторые повторяются)
	requestIDs := []string{
		"apple", "banana", "cherry", "apple",
		"date", "banana", "fig", "grape", 
		"apple", "kiwi", "lime", "banana",
	}

	// Запускаем обработку в отдельных горутинах
	var wg sync.WaitGroup
	for i, id := range requestIDs {
		wg.Add(1)
		go func(requestNum int, requestID string) {
			defer wg.Done()
			fmt.Printf("Request %d for %s started\n", requestNum, requestID)
			subscriber.Consume(requestID)
			fmt.Printf("Request %d for %s completed\n", requestNum, requestID)
		}(i, id)
	}

	wg.Wait()
	fmt.Println("All requests processed")
}

// Пример обработчика
func handler1(id string) error {
	fmt.Printf("[%s] Validation started\n", id)
	time.Sleep(100 * time.Millisecond) // Имитация работы
	fmt.Printf("[%s] Validation completed\n", id)
	return nil
}
```

### Решение

```go

package main  
  
import "sync"  
  
type Handler func(id string) error  
  
type Subscriber struct {  
    handlers []Handler  
    // мьютекс для работы с locks  
    mu       sync.Mutex  
    // ключ: id события, знчение: мьютекс для обработки события  
    locks    map[string]*sync.Mutex  
}  
  
func NewSubscriber(handlers []Handler) *Subscriber {  
    return &Subscriber{  
        handlers: handlers,  
        locks:    make(map[string]*sync.Mutex),  
    }  
}  
  
func (s *Subscriber) Consume(id string) error {  
    // Получаем или создаем мьютекс для конкретного ID  
    mutex := s.getMutex(id)  
      
    // Блокируем обработку для этого ID  
    mutex.Lock()  
    defer mutex.Unlock()  
      
    // Выполняем цепочку обработчиков  
    for _, h := range s.handlers {  
        if err := h(id); err != nil {  
            return err  
        }  
    }  
    return nil  
}  
  
func (s *Subscriber) getMutex(id string) *sync.Mutex {  
    s.mu.Lock()  
    defer s.mu.Unlock()  
      
    if mutex, exists := s.locks[id]; exists {  
        return mutex  
    }  
      
    mutex := &sync.Mutex{}  
    s.locks[id] = mutex  
    return mutex  
}  
```

Можно улучшить решение за счет шардирования - это повысит пропускную способность, но осложнит код. Текущего варианта в большинстве случаев достаточно.
